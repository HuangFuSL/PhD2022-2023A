    \paragraph*{1.48} 由题，格路法到达$x + y = k (x\geq y)$的点的总方案数量为

    \begin{equation}
        \sum_{i=1}^{p} \co ki
    \end{equation}

    式中$p$为$x + y = k$落在区域$x\geq 0, y\leq n, y\geq x$内的整数点的数量。

    \begin{enumerate}
        \item[$k\leq n$时] 有$p = \left\lfloor k/2 \right\rfloor$
        \item[$k> n$时] 有$p = n - \left\lfloor k/2 \right\rfloor$ 
    \end{enumerate}
